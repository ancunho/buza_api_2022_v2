listToTree.forEach(item => {
  delete item.children;
});

let map = {};
listToTree.forEach(item => {
  map[item.menuId] = item;
});
listToTree.forEach(item => {
  let parent = map[item.menuParentId];
  if (parent) {
    (parent.children || (parent.children = [])).push(item);
  } else {
    result.push(item);
  }
});
